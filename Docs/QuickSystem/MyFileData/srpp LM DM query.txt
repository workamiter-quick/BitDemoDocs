(select dm.Code as CodeSQL,dm.Caption as CaptionSQL, l.dm_code as CodePGSQL, l.dis_munic as CaptionPGSQL

insert into LMFinalData
select CodeSQL,CaptionSQL, CaptionPGSQL, CodePGSQL  from 
(select lm.Code as CodeSQL,lm.Caption as CaptionSQL, loc.lm_code as CodePGSQL, loc.loc_munic as CaptionPGSQL
from LMFinal lm left outer join 
(select distinct lm_code, loc_munic from Location) as loc on loc.lm_code = lm.Code) as d order  by CodeSQL
