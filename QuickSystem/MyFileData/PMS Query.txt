SELECT * FROM sys.columns WHERE object_id = OBJECT_ID('dbo.P_Data_Search') and column_id > 3

select ', FLD' + ColumID + ' as [' + Caption + ']', Caption, * from 
(SELECT SUBSTRING(Name, 4, 70) as ColumID FROM sys.columns WHERE object_id = OBJECT_ID('dbo.P_Data_Search') and column_id > 3) as a
inner join P_Attributes pa on PA.AttributeID = a.ColumID
